### map 复合使用：map<int, map<int, int>> 完全可行！
map 支持**嵌套复合使用**（即 value 可以是另一个 map/容器），`map<int, map<int, int>>` 是合法且非常常用的写法，本质是「键值对的 value 本身又是一个键值对容器」。

#### 一、核心理解：map 的 value 可以是任意类型
map 的模板定义是 `map<Key, T>`，其中：
- `Key`：键的类型（需支持比较，如int/string等）；
- `T`：值的类型——可以是int/string，也可以是vector、set，甚至另一个map。

`map<int, map<int, int>>` 的含义：
- 外层 map：键是 `int` 类型，值是「`map<int, int>` 类型」；
- 内层 map：键是 `int` 类型，值是 `int` 类型；
- 整体可理解为「二维键值映射」（类似二维数组，但键不必是连续的）。

#### 二、使用示例（直观演示）
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    // 定义复合map：外层键是“学生ID”，内层键是“科目ID”，值是“分数”
    map<int, map<int, int>> student_score;
    
    // 1. 插入数据：学生101，科目2，分数95
    student_score[101][2] = 95;
    // 2. 插入数据：学生101，科目3，分数88
    student_score[101][3] = 88;
    // 3. 插入数据：学生102，科目2，分数90
    student_score[102][2] = 90;
    
    // 遍历复合map
    for (auto& [stu_id, score_map] : student_score) {
        cout << "学生" << stu_id << "的成绩：";
        for (auto& [sub_id, score] : score_map) {
            cout << "科目" << sub_id << "=" << score << " ";
        }
        cout << endl;
    }
    
    // 访问指定数据：学生101的科目2分数
    cout << "学生101的科目2分数：" << student_score[101][2] << endl;
    
    return 0;
}

## 最终AC代码
```cpp
#include <bits/stdc++.h>
#define ios ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
using namespace std;

// P3613 【深基15.例2】寄包柜
int main()
{
	ios;
	
	int n,q;    //寄包柜个数和询问次数
	cin>>n>>q;
	
	map<int, map<int, int>> mp;
	while(q--){
		int op;
		cin>>op;
		
		if(op==1){
			int i, j, k;    //第 i 个柜子的第 j 个格子存入物品 k
			cin>>i>>j>>k;
			mp[i][j]=k; 
			continue;
		}
		if(op==2){
			int i, j;    //查询第 i 个柜子的第 j 个格子
			cin>>i>>j;
			cout<<mp[i][j]<<'\n';
			continue;
		}
	}
    return 0;
}
```
